
1) pwd - показывает текущий каталог.

 
2) mkdir название_папки -  создать папку.
		mkdir  task2

   mkdir название_папки название_папки название_папки -  создать 3 папки.
		$ mkdir task1 task2 task3


3) cd путь - зайти в папку 
		cd g:/QA/git/theory/task1 - зайти  в папку task1.


4) touch название файла - создать файл.
	touch 2.txt

   touch 1.txt 2.txt 3.txt 1.json 2.json - создать 5 файлов.



5) cat название_файла- выводит содержимое файла.
	cat 1.txt


6) ls -показывает, что находится в каталоге.
   ls -f - в том числе скрытые.


7) nano название_файла - открыть файл.

8) CTRL+0, CTRL+x сохранить,выйти файл.



9) cd ..- выйти из папки на уровень выше.

 
10) mv название.расширение /путь куда перемещаем.- переместить файл.

	mv 1.txt 2.txt /g/QA/git  -переместить несколько файлов.



11) cp название файла /путь- копирование файла в другую папку.

     cp 1.txt 2.txt /g/QA/git - копирование несколькоих файлов.



12) find название_файла.расширение   - поиск файла по имени.
	find 2.txt


13) head название файла- вывод первых 10 строк файла.
	head 1.txt 
    head -n название_файла
	head -3 1.txt выведет три первые строки.


14) tail название_файла- выводит 10 последних строк.
    tail-n название_файла- вывод -n строк.
		 tail -3 1.txt -выведет последние 3 строки.


15)date- выводит дару и время


16) less название_файла- просмотр файла .
	less 123.txt
    внутренние команды 
		ctrl-+ увеличить уменьшить масштаб.
		enter   прокрутка текста на 1 строку.



17) grep -w "слово" имя_файла- поиск  слова в файле.
		grep -w "log" 1.txt-  выводит строку, в которой есть шаблон.
		grep -n "поиск" 123.txt- выводит строку и ее номер в которой есть шаблон.
		grep -c "cp" 123.txt -показывает  количество вхождений.
		grep -i "cp" 123.txt- не учитывает регистр.
		grep -v "cp" 123.txt- не показывает строки в которых есть шаблон.
